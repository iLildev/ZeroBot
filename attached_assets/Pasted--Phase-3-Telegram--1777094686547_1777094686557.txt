# 🧱 سنبني 3 مكونات مترابطة: 
## 1. Gateway (FastAPI)  
- نقطة دخول كل Updates من Telegram 
## 2. Delivery System 
- يرسل الـ Update إلى البوت الصحيح
## 3. Wake Buffer 
- يحل مشكلة “الرسالة الضائعة عند الإيقاظ” 
# ⚠️ قبل ما نكتب الكود — نقطة تصميم مهمة
 
Telegram لا يتحدث مع:
- subprocess مباشرة ❌ 
بل مع:
- Webhook endpoint واحد ✔️  
# 🧠 الحل المعماري
 `Telegram    ↓ Gateway (FastAPI)    ↓ Delivery Manager    ↓ Bot (local port) `  
# 🧩 1️⃣ `core/delivery.py`
## 📌 مسؤول عن إرسال التحديث للبوت
 `# core/delivery.py  import httpx   class DeliveryManager:     def __init__(self):         self.client = httpx.AsyncClient(timeout=5.0)      async def forward(self, port: int, update: dict):         url = f"http://127.0.0.1:{port}/webhook"          try:             await self.client.post(url, json=update)         except Exception as e:             raise RuntimeError(f"Delivery failed: {e}") `  
# 🌐 2️⃣ `core/gateway.py`
 
## 📌 FastAPI Webhook Router
 `# core/gateway.py  from fastapi import FastAPI, Request, HTTPException from sqlalchemy.ext.asyncio import AsyncSession from sqlalchemy import select  from database.models import Bot from core.delivery import DeliveryManager   app = FastAPI() delivery = DeliveryManager()   @app.post("/webhook/{bot_id}") async def handle_update(bot_id: str, request: Request):     update = await request.json()      async with AsyncSession() as session:         result = await session.execute(             select(Bot).where(Bot.id == bot_id)         )         bot = result.scalar_one_or_none()          if not bot:             raise HTTPException(404, "Bot not found")          if not bot.is_active:             raise HTTPException(403, "Bot inactive")          await delivery.forward(bot.port, update)      return {"status": "ok"} `  
# 💤 3️⃣ Wake Buffer (مهم جدًا)
 
## 📌 حل مشكلة:
الرسالة تصل قبل أن يستيقظ البوت 
  `# core/wake_buffer.py  import asyncio from collections import defaultdict   class WakeBuffer:     def __init__(self):         self.buffer = defaultdict(list)      async def add(self, bot_id: str, update: dict):         self.buffer[bot_id].append(update)      async def flush(self, bot_id: str):         updates = self.buffer.pop(bot_id, [])         return updates   wake_buffer = WakeBuffer() `  
# 🔗 تعديل Gateway لاستخدام Wake Buffer
 
## ✏️ داخل `handle_update`
 `if bot.is_hibernated:     await wake_buffer.add(bot_id, update)      # لاحقًا سنضيف:     # await orchestrator.wake_bot(bot_id)      return {"status": "buffered"} `  
# 🧪 4️⃣ تعديل Template (مهم جدًا)
 
## 📁 `templates/base_template/main.py`
 
يجب أن يدعم Webhook:
 `from aiohttp import web import os import json  routes = web.RouteTableDef()  @routes.post("/webhook") async def webhook_handler(request):     data = await request.json()      print("Received update:", data)      return web.Response(text="ok")   app = web.Application() app.add_routes(routes)  port = int(os.getenv("BOT_PORT", 8080))  web.run_app(app, port=port) `
